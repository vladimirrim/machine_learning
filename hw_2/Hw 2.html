1 a)¶

Breakpoint $k = 2$, because for any 2 points we can't get pair (1, -1) due to $x_1 < x_2$ and hence if $x_1 > a$ then $x_2$ also greater than $a$

VC dimension $= k - 1 = 1$

Growth function $m_H(k) = k + 1$ because for k points our hypothesis splits them in two halfs and we can choose from $k + 1$ places between points to set $a$

1 b)¶

Breakpoint $k = 3$ : We can't get all tuples from 3 points because if the smallest and the largest ones lie in the interval then the one in the middle also lies in it and hence we can't get tuple (1, -1, 1)

We can get all pairs for $k = 2$ with points $x$ and $ x + \frac{b - a}{2}$

VC dimension $= 2$

$m_H(k) = \binom{k + 1}{2} + 1$, because now we need to choose two different places between $k$ points and also there is variant where all of the points are not in the interval.

class PokePerc:
    def __init__(self, epochs, size):
        self.epochs = epochs
        self.w = np.random.uniform(size=size)
        
    def score(self, X, y):
        return np.mean((self.predict(X) == y))
        
    def predict(self, X):
        return (X @ self.w > 0).astype(int)
    
    def fit(self, X_train, y_train, X_test, y_test):
        best_w = None
        best_score = -1
        for epoch in range(self.epochs):
            for i in range(X_train.shape[0]):
                predicted = self.predict([1] + list(X_train[i]))
                y_apply = 2 * (int(y_train[i] == 1)) - 1
                if predicted != y_train[i]:
                    self.w += y_apply * np.array([1] + list(X_train[i]))
                    current_score = self.score([ [1] + list(x) for x in X_test], y_test)
                    if current_score > best_score:
                        best_score = current_score
                        best_w = np.copy(self.w)
        self.w = best_w

class LogisticRegression:
    def __init__(self, epochs, lr=0.1, batch_size=32):
        self.epochs = epochs
        self.lr = lr
        self.batch_size = batch_size
        self.w = None
        from scipy.stats import logistic
        self.sigmoid = logistic.cdf
        self.labels = None 
        
    def score(self, X, y):
        return np.mean((self.predict(X) == y))
        
    def to_user(self, y):
        y = np.copy(y).astype(object)
        y[y == -1] = self.labels[-1]
        y[y == 1] = self.labels[1]
        return y
    
    def to_minus_plus(self, y):
        y = np.copy(y)
        y[y == self.labels[-1]] = -1
        y[y == self.labels[1]] = 1
        return y
    
    def predict(self, X):
        predicted = (self.sigmoid(X @ self.w) > 0.5).astype(int)
        predicted[predicted == 0] = -1
        return self.to_user(predicted) 
    
    def fit(self, X, y):
        sorted_labels = np.sort(np.unique(y))
        self.labels = {-1: sorted_labels[0], 1: sorted_labels[1]}
        user_y = y
        y = self.to_minus_plus(y)
        np.random.seed(1337)
        self.w = np.zeros(X.shape[1])
        acc = []
        for epoch in range(self.epochs):
            batch_grad = 0
            for i in range(X.shape[0]): 
                batch_grad += y[i] * X[i] / (1 + np.e ** (y[i] * self.w @ X[i]))
                if i % self.batch_size == 0 and i != 0:
                    self.w -= self.lr * (-1 / self.batch_size * batch_grad)
                    batch_grad = 0
            acc.append(self.score(X, user_y))
        return acc

import os
from skimage import io

def change_str_labels(origin_labels):
    labels = np.copy(origin_labels)
    unique_labels = np.sort(np.unique(np.array(origin_labels)))
    for i in range(unique_labels.shape[0]):
        labels[labels == unique_labels[i]] = i
    return labels.astype(float)

root_path = 'notMNIST_small'
origin_labels = []
images = []
for label in os.listdir(root_path):
    for root, _, files in os.walk(os.path.join(root_path, label)):
        origin_labels.extend([label for _ in files])
        for file in files:
            images.append(io.imread(os.path.join(root, file)))
images = np.array(images).reshape(-1, 28, 28, 1)
labels = change_str_labels(origin_labels)
