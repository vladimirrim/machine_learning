from sklearn.metrics import r2_score
from sklearn.preprocessing import PolynomialFeatures


def plot_regression_func(X, y, y_pred, degree, title):
    sorted_X = np.argsort(X, axis=0)
    plt.plot(X[sorted_X].reshape(X.shape), y_pred[sorted_X].reshape(y.shape), linewidth=2, color='red')
    plt.scatter(X, y, s=50, cmap='viridis')
    plt.title(title)
    plt.show()

def plot_regression(model, degree, title, X_train, X_val, y_train, y_val):
    y_pred, _ = calc_r2_score(model, degree, X_train, X_val, y_train, y_val)
    plot_regression_func(np.array(list(X_train) + list(X_val)), 
                         np.array(list(y_train) + list(y_val)), y_pred, degree, title + ' Degree: ' + str(degree))
    
def calc_r2_score(model, degree, X_train, X_val, y_train, y_val):
    model.fit(PolynomialFeatures(degree=degree).fit_transform(X_train), y_train)
    y_pred = model.predict(PolynomialFeatures(degree=degree).fit_transform(X_val))
    r2 = r2_score(y_val, y_pred)
    print('R2 score: ' + str(r2))
    return model.predict(PolynomialFeatures(degree=degree).fit_transform(np.array(list(X_train) + list(X_val)))), r2

def print_r2_score(title, model, degree, X_train, X_val, y_train, y_val):
    print(title + ' Degree: ' + str(degree))
    calc_r2_score(model, degree, X_train, X_val, y_train, y_val)

class LinearRegression:
    def fit(self, X, y):
        self.w = np.linalg.inv(np.transpose(X) @ X) @ np.transpose(X) @ y
    
    def predict(self, X):
        return X @ self.w

class Ridge:
    def __init__(self, alpha):
        self.alpha = alpha
        
    def fit(self, X, y):
        self.w = np.linalg.inv(np.transpose(X) @ X + self.alpha * np.identity(len(X[0]))) @ np.transpose(X) @ y
        
    def predict(self, X):
        return X @ self.w

from sklearn import linear_model

def plot_lasso_parameters(degree, title, X_train, X_val, y_train, y_val):
    alpha = find_lasso_parameters(degree, X_train, X_val, y_train, y_val)
    plot_regression(linear_model.Lasso(alpha=alpha), degree, title, noisysine_X_train,
                    noisysine_X_val, noisysine_y_train, noisysine_y_val)
    
def find_lasso_parameters(degree, X_train, X_val, y_train, y_val):
    alpha_ridge = [1e-15, 1e-10, 1e-8, 1e-4, 1e-3,1e-2, 1, 5, 10, 20]
    r2_scores = []
    models = []
    for alpha in alpha_ridge:
        model = linear_model.Lasso(alpha=alpha)
        models.append(model)
        _, r2 = calc_r2_score(model, degree, X_train, X_val, y_train, y_val)
        r2_scores.append(r2)
    best = np.argmax(r2_scores)
    print('Best alpha: ' + str(alpha_ridge[best]) + ' Best r2 score: ' + str(r2_scores[best]))
    print('Used features: ' + str((np.sum(models[best].coef_ != 0))))
    return alpha_ridge[best]

def print_lasso_parameters(degree, title, X_train, X_val, y_train, y_val):
    print(title)
    alpha = find_lasso_parameters(degree, X_train, X_val, y_train, y_val)

import random

def crossover(gene1, gene2):
    gene1 = gene1.copy()
    gene2 = gene2.copy()
    idx = range(len(gene1))
    n = len(gene1)
    i1, i2 = random.sample(idx, 2)
    residue = range(n - max(i1, i2))
    length = random.sample(residue, 1)[0]
    
    to_gene1 = []
    for i in range(n):
        if gene2[i].tolist() not in gene1[i1: i1 + length].tolist():
            to_gene1.append(gene2[i].copy())
    to_gene2 = []
    for i in range(n):
        if gene1[i].tolist() not in gene2[i2: i2 + length].tolist():
            to_gene2.append(gene1[i].copy())
            
    for i in range(i1):
        gene1[i] = to_gene1[i]
        
    for i in range(i1 + length, n):
        gene1[i] = to_gene1[i - length]
    
    for i in range(i2):
        gene2[i] = to_gene2[i]
        
    for i in range(i2 + length, n):
        gene2[i] = to_gene2[i - length]
        
    return gene1, gene2
    
def mutate(gene):
    gene = gene.copy()
    mutationChance = 0.05
    idx = range(len(gene))
    if np.random.uniform() < mutationChance:
        i1, i2 = random.sample(idx, 2)
        start = min(i1,i2)
        end = max(i1, i2)
        gene[start: end] =  gene[start: end][::-1]
    return gene
